    <script>
        const terminalContent = document.getElementById('terminal-content');

        // 출력할 로그 메시지 데이터베이스
        const logs = [
            "Initializing neural handshake...",
            "Validating shard #89X blocks...",
            "Optimizing gas routes (Layer 2)...",
            "AI Trust Score updated: 99.2%",
            "Burn protocol executing...",
            "Wallet signature verified.",
            "Cross-chain bridge connected.",
            "Syncing with mainnet nodes...",
            "Quantum encryption active.",
            "Governance proposal indexed.",
            "Detecting MEV bots... Blocked.",
            "Asset liquidity rebalancing...",
            "Verifying ZK-proofs..."
        ];

        function addLog() {
            // 랜덤 로그 선택
            const logMessage = logs[Math.floor(Math.random() * logs.length)];
            
            // 현재 시간 포맷팅 (HH:MM:SS)
            const now = new Date();
            const timeString = now.toLocaleTimeString('en-US', { hour12: false });

            // 로그 엘리먼트 생성
            const logEntry = document.createElement('div');
            logEntry.className = 'animate-fade-in'; // Tailwind 설정에 정의된 애니메이션
            
            // HTML 구조: [시간] > 메시지 (색상 적용)
            logEntry.innerHTML = `
                <span class="text-gray-500 mr-1">[${timeString}]</span> 
                <span class="text-gray-400 mr-1">></span> 
                <span class="text-brand-cyan drop-shadow-[0_0_2px_rgba(0,240,255,0.5)]">${logMessage}</span>
            `;

            // 컨테이너에 추가
            terminalContent.appendChild(logEntry);

            // 자동 스크롤 (항상 하단 유지)
            terminalContent.scrollTop = terminalContent.scrollHeight;

            // 성능 최적화: 로그가 너무 많이 쌓이면 상단 로그 제거 (최대 20개 유지)
            if (terminalContent.children.length > 20) {
                terminalContent.removeChild(terminalContent.firstChild);
            }

            // 다음 로그 출력 시간 랜덤 설정 (0.5초 ~ 2.5초 사이)
            const nextInterval = Math.random() * 2000 + 500;
            setTimeout(addLog, nextInterval);
        }

        // 최초 실행
        addLog();
    </script>
