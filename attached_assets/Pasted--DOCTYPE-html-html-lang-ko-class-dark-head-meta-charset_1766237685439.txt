    <script>
        // --- 1. Real-time Preview ---
        function updatePreview() {
            const name = document.getElementById('input-name').value || 'Token Name';
            const symbol = document.getElementById('input-symbol').value || 'SYMBOL';
            const supply = document.getElementById('input-supply').value || '0';

            document.getElementById('card-name').innerText = name;
            document.getElementById('card-symbol').innerText = symbol.toUpperCase();
            document.getElementById('card-icon').innerText = symbol.charAt(0).toUpperCase() || 'T';
            
            // Format supply with commas
            const formattedSupply = parseFloat(supply).toLocaleString('en-US');
            document.getElementById('card-supply').innerText = formattedSupply;
        }

        // --- 2. Fake Deployment Process ---
        function deployToken() {
            const consoleDiv = document.getElementById('console-output');
            const logsDiv = document.getElementById('console-logs');
            const btn = document.querySelector('button[onclick="deployToken()"]');
            
            // Show Console
            consoleDiv.classList.remove('hidden');
            btn.disabled = true;
            btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Deploying...';
            btn.classList.add('opacity-70', 'cursor-not-allowed');

            logsDiv.innerHTML = ''; // Clear logs
            
            const steps = [
                { msg: "> Initializing TBC-20 Factory...", delay: 500 },
                { msg: "> Compiling Smart Contract Bytecode...", delay: 1200 },
                { msg: "> Connecting to TBURN Mainnet (ChainID: 882)...", delay: 2000 },
                { msg: "> Estimating Gas... <span class='text-tburn-success'>0.00045 TB (Ultra Low)</span>", delay: 2800 },
                { msg: "> Signing Transaction with Wallet...", delay: 3500 },
                { msg: "> Broadcasting to Mempool...", delay: 4200 },
                { msg: "> <span class='text-tburn-success'>[SUCCESS]</span> Block #21,334,102 Confirmed (0.4s)", delay: 5000 },
                { msg: "> Contract Address: <span class='text-tburn-code'>0x71C...9A21</span>", delay: 5200 },
                { msg: "> Verifying Source Code on Explorer...", delay: 6000 },
                { msg: "> <span class='text-tburn-success font-bold'>Token Deployed Successfully! üöÄ</span>", delay: 6500 }
            ];

            let cumulativeDelay = 0;

            steps.forEach(step => {
                cumulativeDelay = step.delay;
                setTimeout(() => {
                    const p = document.createElement('div');
                    p.innerHTML = `<span class="text-slate-500">[${new Date().toLocaleTimeString()}]</span> ${step.msg}`;
                    logsDiv.appendChild(p);
                    logsDiv.scrollTop = logsDiv.scrollHeight; // Auto scroll
                }, step.delay);
            });

            // Reset Button
            setTimeout(() => {
                btn.disabled = false;
                btn.innerHTML = '<i class="fas fa-check"></i> Done';
                btn.classList.remove('from-blue-600', 'to-blue-500');
                btn.classList.add('from-green-600', 'to-green-500');
            }, cumulativeDelay);
        }

        // --- 3. Theme Toggle ---
        function toggleTheme() {
            const html = document.documentElement;
            const themeIcon = document.getElementById('theme-icon');
            if (html.classList.contains('dark')) {
                html.classList.remove('dark');
                themeIcon.classList.remove('fa-sun');
                themeIcon.classList.add('fa-moon');
            } else {
                html.classList.add('dark');
                themeIcon.classList.remove('fa-moon');
                themeIcon.classList.add('fa-sun');
            }
        }
    </script>
