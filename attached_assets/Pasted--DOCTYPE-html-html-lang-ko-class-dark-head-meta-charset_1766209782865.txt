    <script>
        // --- 1. Navigation Logic ---
        function switchSection(sectionId) {
            // Hide all
            document.getElementById('section-nft').classList.add('hidden');
            document.getElementById('section-launchpad').classList.add('hidden');
            document.getElementById('section-gamefi').classList.add('hidden');
            
            // Reset Buttons
            const buttons = ['nft', 'launchpad', 'gamefi'];
            buttons.forEach(btn => {
                const el = document.getElementById('btn-' + btn);
                el.className = "px-4 py-1.5 rounded-full text-sm font-bold text-slate-500 dark:text-gray-400 hover:bg-slate-100 dark:hover:bg-gray-800 transition-all";
            });

            // Show selected & Active Style
            document.getElementById('section-' + sectionId).classList.remove('hidden');
            
            const activeBtn = document.getElementById('btn-' + sectionId);
            let activeClass = "";
            if(sectionId === 'nft') activeClass = "bg-tburn-nft/10 text-tburn-nft ring-1 ring-tburn-nft/50";
            if(sectionId === 'launchpad') activeClass = "bg-tburn-launch/10 text-tburn-launch ring-1 ring-tburn-launch/50";
            if(sectionId === 'gamefi') activeClass = "bg-tburn-game/10 text-tburn-game ring-1 ring-tburn-game/50";
            
            activeBtn.className = "px-4 py-1.5 rounded-full text-sm font-bold transition-all " + activeClass;
        }

        // --- 2. Theme Toggle ---
        function toggleTheme() {
            const html = document.documentElement;
            const themeIcon = document.getElementById('theme-icon');
            if (html.classList.contains('dark')) {
                html.classList.remove('dark');
                themeIcon.classList.remove('fa-sun');
                themeIcon.classList.add('fa-moon');
            } else {
                html.classList.add('dark');
                themeIcon.classList.remove('fa-moon');
                themeIcon.classList.add('fa-sun');
            }
        }

        // --- 3. Countdown Timer (Launchpad) ---
        function updateTimer() {
            // Mock countdown
            const now = new Date();
            const hours = 4 - (now.getHours() % 4);
            const minutes = 60 - now.getMinutes();
            const seconds = 60 - now.getSeconds();
            document.getElementById('countdown').innerText = 
                `${hours}h ${minutes}m ${seconds}s`;
        }
        setInterval(updateTimer, 1000);

    </script>
