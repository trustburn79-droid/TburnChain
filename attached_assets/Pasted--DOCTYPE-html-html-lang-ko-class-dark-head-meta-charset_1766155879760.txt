    <script>
        // --- 1. Theme Toggle ---
        function toggleTheme() {
            const html = document.documentElement;
            const themeIcon = document.getElementById('theme-icon');
            if (html.classList.contains('dark')) {
                html.classList.remove('dark');
                themeIcon.classList.remove('fa-sun');
                themeIcon.classList.add('fa-moon');
            } else {
                html.classList.add('dark');
                themeIcon.classList.remove('fa-moon');
                themeIcon.classList.add('fa-sun');
            }
        }

        // --- 2. Modal Logic ---
        let currentApr = 0;

        function openDelegateModal(name, apr) {
            const modal = document.getElementById('delegate-modal');
            const content = document.getElementById('modal-content');
            
            // Set Data
            document.getElementById('modal-validator-name').innerText = name;
            document.getElementById('modal-validator-apr').innerText = apr;
            document.getElementById('modal-validator-img').src = `https://api.dicebear.com/7.x/identicon/svg?seed=${name.replace(' ','')}`;
            currentApr = apr;
            
            // Reset Input
            document.getElementById('stake-amount').value = '';
            document.getElementById('est-daily').innerText = '0.00';
            document.getElementById('est-monthly').innerText = '0.00';

            // Animation
            modal.classList.remove('hidden');
            setTimeout(() => {
                content.classList.remove('scale-95', 'opacity-0');
                content.classList.add('scale-100', 'opacity-100');
            }, 10);
        }

        function closeDelegateModal() {
            const modal = document.getElementById('delegate-modal');
            const content = document.getElementById('modal-content');
            
            content.classList.remove('scale-100', 'opacity-100');
            content.classList.add('scale-95', 'opacity-0');
            
            setTimeout(() => {
                modal.classList.add('hidden');
            }, 300);
        }

        // --- 3. Calculator Logic ---
        function calculateRewards() {
            const amount = parseFloat(document.getElementById('stake-amount').value) || 0;
            const daily = (amount * (currentApr / 100)) / 365;
            const monthly = daily * 30;

            document.getElementById('est-daily').innerText = daily.toFixed(4);
            document.getElementById('est-monthly').innerText = monthly.toFixed(4);
        }

        function setPercent(percent) {
            const max = 12500; // Mock Max Balance
            const amount = max * (percent / 100);
            document.getElementById('stake-amount').value = amount.toFixed(2);
            calculateRewards();
        }
    </script>
