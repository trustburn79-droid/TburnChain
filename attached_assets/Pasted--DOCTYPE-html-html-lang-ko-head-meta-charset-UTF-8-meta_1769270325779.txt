        // HEALTH RINGS
        // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
        function updateHealthRing(id, value, max) {
            const circumference = 2 * Math.PI * 34; // r=34
            const percent = Math.min(value / max, 1);
            const dashArray = `${percent * circumference} ${circumference}`;
            $(id).setAttribute('stroke-dasharray', dashArray);
        }

        // BLOCKS TABLE
        // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
        function generateBlocks() {
            const producers = ['val-kr-seoul-01', 'val-us-east-03', 'val-eu-frank-02', 'val-ap-tokyo-05', 'val-sg-west-04'];
            state.blocks = [];
            for (let i = 0; i < 8; i++) {
                state.blocks.push({
                    height: state.blockHeight - i,
                    txs: Math.floor(2000 + Math.random() * 5000),
                    producer: producers[i % producers.length],
                    age: i * 3 + Math.floor(Math.random() * 2)
                });
            }
        }

        function renderBlocks() {
            const tbody = $('blocks-table-body');
            tbody.innerHTML = state.blocks.map(b => `
                <tr>
                    <td class="block-height">#${formatNumber(b.height)}</td>
                    <td class="block-txs">${formatNumber(b.txs)}</td>
                    <td class="block-producer">${b.producer}</td>
                    <td class="block-time">${b.age}s ago</td>
                </tr>
            `).join('');
        }

        // VALIDATORS
        // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
        function generateValidators() {
            const regions = ['Korea', 'US East', 'EU West', 'Singapore', 'Japan', 'Australia'];
            state.validators = [];
            for (let i = 0; i < 8; i++) {
                state.validators.push({
                    name: `validator-${String(i + 1).padStart(2, '0')}`,
                    region: regions[i % regions.length],
                    stake: (500000 + Math.random() * 2000000).toFixed(0),
                    status: Math.random() > 0.1 ? 'active' : 'syncing'
                });
            }
        }

        function renderValidators() {
            const grid = $('validators-grid');
            const activeCount = state.validators.filter(v => v.status === 'active').length;
            $('validator-count').textContent = `${activeCount} Active`;
            
            grid.innerHTML = state.validators.map(v => `
                <div class="validator-item">
                    <div class="validator-status ${v.status}"></div>
                    <div class="validator-info">
                        <div class="validator-name">${v.name}</div>
                        <div class="validator-region">${v.region}</div>
                    </div>
                    <div class="validator-stake">${formatNumber(parseInt(v.stake))} TBURN</div>
                </div>
            `).join('');
        }

        // UPDATE LOOP
        // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
        function updateStats() {
            // Simulate data changes
            const tpsBase = 150000 + Math.sin(Date.now() / 5000) * 10000;
            state.tps = Math.floor(tpsBase + (Math.random() * 10000 - 5000));
            state.blockHeight += Math.floor(1 + Math.random() * 3);
            state.dailyTxs += Math.floor(state.tps * (0.3 + Math.random() * 0.5));
            state.finality = 1.8 + Math.random() * 0.5;
            state.rpcLatency = 35 + Math.random() * 25;
            state.crossShard = 90 + Math.random() * 8;
            state.memEfficiency = 85 + Math.random() * 10;
            state.activePeers = 2800 + Math.floor(Math.random() * 100);

            // Update DOM
            $('tps-value').textContent = formatNumber(state.tps);
            $('block-value').textContent = formatNumber(state.blockHeight);
            $('tx-value').textContent = formatNumber(state.dailyTxs);
            $('uptime-value').textContent = state.uptime.toFixed(2) + '%';

            // Update health metrics
            $('finality-value').textContent = state.finality.toFixed(1) + 's';
            $('rpc-value').textContent = Math.floor(state.rpcLatency) + 'ms';
            $('shard-value').textContent = Math.floor(state.crossShard) + '%';
            $('mem-value').textContent = Math.floor(state.memEfficiency) + '%';
            $('peer-value').textContent = formatNumber(state.activePeers);

            // Update health rings (normalized to 100)
            updateHealthRing('health-finality', (3 - state.finality) / 3 * 100, 100);
            updateHealthRing('health-rpc', (100 - state.rpcLatency) / 100 * 100, 100);
            updateHealthRing('health-shard', state.crossShard, 100);
            updateHealthRing('health-mem', state.memEfficiency, 100);
            updateHealthRing('health-peer', Math.min(state.activePeers / 30, 100), 100);

            // Update charts
            tpsData.push(state.tps);
            tpsData.shift();
            peakData.push(155000 + Math.random() * 10000);
            peakData.shift();
            finalityData.push(state.finality);
            finalityData.shift();
            rpcData.push(state.rpcLatency);
            rpcData.shift();

            tpsChart.update('none');
            latencyChart.update('none');

            // Update blocks
            generateBlocks();
            renderBlocks();

            // Update clock
            $('clock').textContent = formatTime();
            $('last-update').textContent = formatTimestamp();

            // Update block rate
            $('block-rate').textContent = `+${1 + Math.floor(Math.random() * 3)}/s`;
        }

        // INITIALIZATION
        // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
        function init() {
            createParticles();
            initCharts();
            generateBlocks();
            generateValidators();
            renderBlocks();
            renderValidators();
            updateStats();

            // Start update loop
            setInterval(updateStats, 1000);

            // Occasionally update validators
            setInterval(() => {
                generateValidators();
                renderValidators();
            }, 10000);
        }

        // Start
        document.addEventListener('DOMContentLoaded', init);
    </script>
