    // Countdown Timer
    function updateCountdown() {
      const endDate = new Date('2025-02-15T00:00:00');
      const now = new Date();
      const diff = endDate - now;

      if (diff > 0) {
        const days = Math.floor(diff / (1000 * 60 * 60 * 24));
        const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
        const seconds = Math.floor((diff % (1000 * 60)) / 1000);

        document.getElementById('days').textContent = days.toString().padStart(2, '0');
        document.getElementById('hours').textContent = hours.toString().padStart(2, '0');
        document.getElementById('minutes').textContent = minutes.toString().padStart(2, '0');
        document.getElementById('seconds').textContent = seconds.toString().padStart(2, '0');
      }
    }

    setInterval(updateCountdown, 1000);
    updateCountdown();

    // Calculate Allocation
    function calculateAllocation() {
      const amount = parseFloat(document.getElementById('allocationInput').value) || 0;
      const tokenPrice = 0.020;

      let bonusPercent = 0;
      if (amount >= 50000) bonusPercent = 5;
      else if (amount >= 10000) bonusPercent = 3;
      else if (amount >= 1000) bonusPercent = 1;

      const baseTokens = amount / tokenPrice;
      const bonusTokens = baseTokens * (bonusPercent / 100);
      const totalTokens = baseTokens + bonusTokens;
      const tgeTokens = totalTokens * 0.15;

      document.getElementById('tokenAmount').textContent = formatNumber(baseTokens) + ' TBURN';
      document.getElementById('bonusAmount').textContent = bonusPercent > 0 ? '+' + formatNumber(bonusTokens) + ' TBURN' : '-';
      document.getElementById('tgeAmount').textContent = formatNumber(tgeTokens) + ' TBURN';
    }

    function formatNumber(num) {
      return Math.floor(num).toLocaleString();
    }

    // Set Allocation

      document.getElementById('allocationInput').value = amount;

      calculateAllocation();
