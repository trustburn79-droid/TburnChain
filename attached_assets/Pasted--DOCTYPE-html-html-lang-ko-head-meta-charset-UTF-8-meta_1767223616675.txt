    <script>
        function calculateRewards() {
            // Inputs
            const amount = parseFloat(document.getElementById('stakeAmount').value) || 0;
            const months = parseInt(document.getElementById('duration').value);
            
            // Logic: Base APY 7.24% + 0.1% bonus per month locked
            const baseAPY = 0.0724;
            const bonusAPY = (months * 0.001); 
            const totalAPY = baseAPY + bonusAPY;
            
            const annualEarnings = amount * totalAPY;
            const periodEarnings = annualEarnings * (months / 12);
            
            const monthly = annualEarnings / 12;
            const daily = annualEarnings / 365;

            // UI Updates
            document.getElementById('durationLabel').innerText = months + (months === 1 ? " Month" : " Months");
            
            // Format numbers with commas
            const fmt = (num) => num.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
            
            document.getElementById('totalEarnings').innerText = "+" + fmt(periodEarnings);
            document.getElementById('monthlyEarnings').innerText = fmt(monthly);
            document.getElementById('dailyEarnings').innerText = fmt(daily);
        }

        // Initialize on load
        calculateRewards();
    </script>
