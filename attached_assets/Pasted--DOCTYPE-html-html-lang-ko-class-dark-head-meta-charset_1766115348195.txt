    <script>
        // --- 0. Theme Toggle Logic ---
        function toggleTheme() {
            const html = document.documentElement;
            const themeIcon = document.getElementById('theme-icon');
            
            if (html.classList.contains('dark')) {
                html.classList.remove('dark');
                themeIcon.classList.remove('fa-sun');
                themeIcon.classList.add('fa-moon');
                // Update Chart for Light Mode
                updateChartColor(false);
            } else {
                html.classList.add('dark');
                themeIcon.classList.remove('fa-moon');
                themeIcon.classList.add('fa-sun');
                // Update Chart for Dark Mode
                updateChartColor(true);
            }
        }

        // --- 1. Navigation Logic ---
        function showSection(sectionId) {
            const sections = ['dashboard', 'wallet', 'staking', 'governance', 'network'];
            sections.forEach(id => {
                document.getElementById(id).classList.add('hidden');
            });
            document.getElementById(sectionId).classList.remove('hidden');

            const navItems = document.querySelectorAll('.nav-item');
            navItems.forEach(item => {
                // Reset to default state
                item.className = 'nav-item w-full flex items-center gap-4 px-4 py-3 text-sm font-medium rounded-xl transition-all duration-200 border-l-4 border-transparent text-slate-500 hover:bg-slate-100 dark:text-gray-400 dark:hover:bg-gray-800 hover:text-slate-900 dark:hover:text-white';
                
                // Active state styling
                if(item.getAttribute('onclick').includes(sectionId)) {
                     item.className = 'nav-item w-full flex items-center gap-4 px-4 py-3 text-sm font-medium rounded-xl transition-all duration-200 bg-blue-50 text-tburn-primary border-l-4 border-tburn-primary dark:bg-tburn-cardDark dark:text-white dark:border-tburn-primary shadow-sm';
                }
            });
        }

        // --- 2. Trust Score Chart (Chart.js) ---
        let trustChart;
        
        function initChart() {
            const ctx = document.getElementById('trustScoreChart').getContext('2d');
            const isDark = document.documentElement.classList.contains('dark');
            
            trustChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['Score', 'Remaining'],
                    datasets: [{
                        data: [92, 8],
                        backgroundColor: [
                            '#3B82F6', // Trust Blue
                            isDark ? 'rgba(255, 255, 255, 0.05)' : 'rgba(0, 0, 0, 0.05)'
                        ],
                        borderWidth: 0,
                        borderRadius: 20,
                        cutout: '85%'
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { display: false },
                        tooltip: { enabled: false }
                    },
                    animation: {
                        animateScale: true,
                        animateRotate: true
                    }
                }
            });
        }

        function updateChartColor(isDark) {
            if(trustChart) {
                trustChart.data.datasets[0].backgroundColor[1] = isDark ? 'rgba(255, 255, 255, 0.05)' : 'rgba(0, 0, 0, 0.1)';
                trustChart.update();
            }
        }

        // Initialize Chart on load
        initChart();

        // --- 3. Live Feed Simulation ---
        function updateFeed() {
            const feed = document.getElementById('block-feed');
            if(feed && !document.getElementById('network').classList.contains('hidden')) {
                const currentBlock = 8921000 + Math.floor(Date.now() / 1000);
                const txns = Math.floor(Math.random() * (60000 - 30000) + 30000).toLocaleString();
                const burned = Math.floor(Math.random() * 200);
                
                const newRow = document.createElement('div');
                newRow.className = "flex justify-between text-slate-500 dark:text-gray-300 py-1 border-b border-slate-100 dark:border-gray-800/50 animate-pulse";
                newRow.innerHTML = `
                    <span class="text-slate-800 dark:text-white">Block #${currentBlock}</span>
                    <span>Txns: ${txns}</span>
                    <span class="text-tburn-accent">Burned: ${burned} TB</span>
                    <span class="text-tburn-success">Just now</span>
                `;
                
                feed.prepend(newRow);
                if(feed.children.length > 8) feed.lastElementChild.remove();
            }
        }
        setInterval(updateFeed, 1000);
